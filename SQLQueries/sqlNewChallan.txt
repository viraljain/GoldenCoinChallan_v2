select T1.ItemName, T1.ItemSize, T1.UnitName, iif(TblSizeMap.ParentSize IS NULL, t1.ItemSize, TblSizeMap.ParentSize) AS ItemSize, TblSizeMap.SizeSNo
from
(
	select	LEFT(Material.Name, len(Material.Name) - charindex(' ', reverse(Material.Name), 0)) AS ItemName, 
			RIGHT(Material.Name, iif(charindex(' ', reverse(Material.Name), 0) = 0, 1, charindex(' ', reverse(Material.Name), 0)) - 1) AS ItemSize,
			dbo.Material.Code, Ucode, UnitCode, TblUnit.UnitName
	from dbo.Material
	INNER JOIN
	(
		SELECT        [Code], [FieldValue1] as UnitName
		FROM            [dbo].[Master]
		WHERE        FieldName1 = 'Unit Name'
	) AS TblUnit 
	ON TblUnit.Code = Material.UnitCode
	where Name like '%N-1%'
) as T1
LEFT JOIN
dbo.tblSizeMap AS TblSizeMap 
ON T1.ItemSize = TblSizeMap.ChildSize;
